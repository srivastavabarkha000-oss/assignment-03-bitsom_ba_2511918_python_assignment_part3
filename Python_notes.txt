#Part 3: File I/O, APIs & Exception Handling
# Task 1:File Read and Write Basics
from pathlib import Path

# Part A: Write to File

base = Path(__file__).parent
data_dir = base / "MyFiles"
data_dir.mkdir(exist_ok=True)

print("Working in:", data_dir)

# 1: Write initial code
text_path = data_dir / "python_notes.txt"
with open(text_path, "w", encoding="utf-8") as f:
    f.write("Topic 1: Variables store data. Python is dynamically typed.\n")
    f.write("Topic 2: Lists are ordered and mutable.\n")
    f.write("Topic 3: Dictionaries store key-value pairs.\n")
    f.write("Topic 4: Loops automate repetitive tasks.\n")
    f.write("Topic 5: Exception handling prevents crashes.\n")

print("File written successfully.", text_path)

# 2: Append to the same file (a = append)
with open(text_path, "a", encoding="utf-8") as f:
    f.write("Topic 6: Functions help in code reusablity\n")
    f.write("Topic 7: File handling allows persistent storage")
print("Lines Appended.")


# 3: Read the whole file at once (r = read)
with open(text_path, "r", encoding="utf-8") as f:
    content = f.read()
print("--- File content (read) ---")
print(content)


# Part B: Read from File
try:
    with open(text_path, "r", encoding="utf-8") as f:
        lines = f.readlines()

# Print numbered lines (without \n)
    print("\n--- File Content ---")
    for i, line in enumerate(lines, start=1):
        print(f"{i}. {line.strip()}")

# Count total lines
    total_lines = len(lines)
    print(f"\nTotal number of lines: {total_lines}")

#  Keyword search (case-insensitive)
    keyword = input("\nEnter a keyword to search: ").lower()
    matches = [line.strip() for line in lines if keyword in line.lower()]

    print("\n--- Search Results ---")
    if matches:
        for match in matches:
            print(match)
    else:
        print("No matching lines found for the given keyword.")

except FileNotFoundError:
    print("Error: The file 'python_notes.txt' does not exist.")
except Exception as e:
    print("An error occurred:", e)
